<h1 align="center">常用字符串方法</h1>

```
\\ 查找字符串中对应的字符下标，返回字符下标或 -1；注：参数 `5` 为要查找的起始下标（包含该下标）；该参数为 [0 ~ str.length - 1] 之间的整数；
var str = "Hello world!";
str.indexOf("o") // 4
str.indexOf("o", 5); // 7 ；

\\ 替换指定字符或字符串，返回新字符串，不改变原字符串；
var str = "Hello world!";
str.replace("o", "0") // "hell0 w0rld";

\\ 把小写字母转换为大写，返回转换后的字符串，不改变原字符串；
var str = "Hello world!";
str.toUpperCase()  // "HELLO WORLD!"

\\ 把大写字母转换为小写，返回转换后的字符串，不改变原字符串；
var str = "Hello world!";
str.toLowerCase()  // "hello world!"

\\ 去除字符串前后空格，返回去除空格后的字符串，不改变原字符串；
var str = " Hello world!  ";
str.trim();  // "Hello world!"

\\ 把字符串以指定分隔符分隔成数组，返回分隔后的数组，不改变原字符串；
var str = "Hello world!";
str.split(" ")  // ["Hello", "world!"] 以空格为分隔符

\\ 字符串截取，返回被截取的字符串，不改变原字符串；
substr(start, length)
var str = " Hello world!  ";
str.substr(2, 3)  // "llo"

\\ 字符串拼接
var str="aaa";
console.log(str.concat("sss"))//aaasss
console.log(str)//aaa
```
