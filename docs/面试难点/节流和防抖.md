# [节流和防抖](https://www.cnblogs.com/cc-freiheit/p/10827372.html)

```js
// 防抖 在一个动作结束后一段时间，执行函数（搜索功能）
function debounce(fn, wait) {
  let timer
  return function() {
    if(timer) clearTimeout(timer)

    timer = setTimeout(() => {
      fn.apply(this)
    }, wait)
  }
} 

// 节流 一段时间执行函数(浏览器resize)
function throttle() {
  let timer
  return function () {
    if (!timer) {
      timer = setTimeout(() => {
        fn.apply(this)
        timer = null
      }, wait)
    }
  }
}
```
